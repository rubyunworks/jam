    def append(node_or_nodeset, child)
      ns = node_or_nodeset
      case child
      when ::Nokogiri::XML::Node
        each_node(node_or_nodeset) do |node|
          node << child
        end
      when ::Nokogiri::XML::NodeSet
        child.each do |n|
          each_node(node_or_nodeset) do |node|
            node << n.dup
          end
        end
      else
        append_text(node_or_nodeset, child)        
      end
    end

    def each_node(nodes)
      unless ::Nokogiri::XML::NodeSet === nodes
        nodes = [nodes]
      end
      nodes.each do |node|
        yield(node)
      end
    end
